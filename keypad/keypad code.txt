#define F_CPU 8000000UL
#include <avr/io.h>
#include <util/delay.h>

// -------- LCD COMMAND --------
void lcd_command(unsigned char cmd)
{
	PORTA = cmd;

	PORTB = 0x00;
	_delay_ms(1);

	PORTB = 0x02;
	_delay_ms(1);

	PORTB = 0x00;
	_delay_ms(2);
}

// -------- LCD DATA --------
void lcd_data(unsigned char data)
{
	PORTA = data;

	PORTB = 0x01;
	_delay_ms(1);

	PORTB = 0x03;
	_delay_ms(1);

	PORTB = 0x01;
	_delay_ms(2);
}

// -------- LCD INIT --------
void lcd_init()
{
	DDRA = 0xFF;
	DDRB = 0xFF;

	_delay_ms(20);

	lcd_command(0x38);
	lcd_command(0x0C);
	lcd_command(0x01);
	lcd_command(0x06);
}

// -------- WAIT RELEASE --------
void wait_release()
{
	while ((PIND & 0xF0) != 0xF0);
	_delay_ms(5);
}

// -------- KEYPAD USING LOOP --------
char keypad()
{
	char keys[4][4] = {
		{'7','8','9','A'},
		{'4','5','6','B'},
		{'1','2','3','C'},
		{'C','0','=','D'}
	};

	unsigned char row_values[4] = {0x0E, 0x0D, 0x0B, 0x07};
	unsigned char col_values[4] = {0xE0, 0xD0, 0xB0, 0x70};

	for (int i = 0; i < 4; i++)
	{
		PORTD = (PORTD & 0xF0) | row_values[i];

		if ((PIND & 0xF0) != 0xF0)
		{
			_delay_ms(5);

			for (int j = 0; j < 4; j++)
			{
				if ((PIND & 0xF0) == col_values[j])
				{
					wait_release();
					return keys[i][j];
				}
			}
		}
	}

	return 0;
}

// -------- MAIN --------
int main(void)
{
	lcd_init();

	DDRD = 0x0F;
	PORTD = 0xF0;

	lcd_command(0x80);

	while (1)
	{
		char key = keypad();

		if (key != 0)
		{
			lcd_data(key);
		}
	}
}
