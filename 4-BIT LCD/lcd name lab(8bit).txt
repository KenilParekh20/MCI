#define F_CPU 8000000UL     // Set this to your internal/external clock speed
#include <avr/io.h>
#include <util/delay.h>

// Control Pins on PORTB
#define RS PB0              // LCD Pin 4
#define EN PB1              // LCD Pin 6
#define LCD_CTRL_PORT PORTB
#define LCD_DATA_PORT PORTA // LCD Pins 7-14 connected to PA0-PA7

void LCD_Command(unsigned char cmnd) {
    LCD_DATA_PORT = cmnd;         // Put command on PORTA
    LCD_CTRL_PORT &= ~(1 << RS);  // RS = 0 for command
    LCD_CTRL_PORT |= (1 << EN);   // Enable Pulse High
    _delay_ms(1);
    LCD_CTRL_PORT &= ~(1 << EN);  // Enable Pulse Low
    _delay_ms(1);                 // Wait for command to process
}

void LCD_Char(unsigned char data) {
    LCD_DATA_PORT = data;         // Put character on PORTA
    LCD_CTRL_PORT |= (1 << RS);   // RS = 1 for data
    LCD_CTRL_PORT |= (1 << EN);   // Enable Pulse High
    _delay_ms(1);
    LCD_CTRL_PORT &= ~(1 << EN);  // Enable Pulse Low
    _delay_ms(1);                 // Wait for data to process
}

void LCD_Init(void) {
    DDRA = 0xFF;                  // Set PORTA as Output (Data)
    DDRB |= (1 << RS) | (1 << EN);// Set PB0 and PB1 as Output (Control)
    
    _delay_ms(20);                // LCD Power-on delay
    
    LCD_Command(0x38);            // 8-bit mode, 2 lines, 5x7 dots
    LCD_Command(0x0C);            // Display ON, Cursor OFF
    LCD_Command(0x06);            // Entry mode: Auto-increment cursor
    LCD_Command(0x01);            // Clear display
    _delay_ms(2);
}

void LCD_String(char *str) {
    for(int i=0; str[i] != 0; i++) {
        LCD_Char(str[i]);
    }
}

int main(void) {
    LCD_Init();                   // Initialize the LCD
    
    // Move to Row 1, Column 5 to center "Meet"
    LCD_Command(0x84);            
    LCD_String("Meet");

    while(1) {
        // Keep the display static
    }
}
