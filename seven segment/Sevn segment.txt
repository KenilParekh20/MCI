//if single button is pressed than is should count 0-9 and two is pressed than it should count 0-F
#define F_CPU 8000000UL
#include <avr/io.h>
#include <util/delay.h>

// Seven segment values (common cathode)
int hex[16] =
{
	0x3F, //0
	0x06, //1
	0x5B, //2
	0x4F, //3
	0x66, //4
	0x6D, //5
	0x7D, //6
	0x07, //7
	0x7F, //8
	0x6F, //9
	0x77, //A
	0x7C, //B
	0x39, //C
	0x5E, //D
	0x79, //E
	0x71  //F
};

int main(void)
{
	DDRA = 0xFF;   // PORTA output for SSD
	DDRB = 0x00;   // PORTB input for button1
	DDRD = 0x00;   // PORTD input for button2

	PORTB = 0xFF;  // Enable pull-up resistor
	PORTD = 0xFF;  // Enable pull-up resistor

	while (1)
	{
		int b1 = !(PINB & 1);   // read button1
		int b2 = !(PIND & 32);   // read button2

		_delay_ms(30);   // debounce delay


		// Both buttons not pressed
		if(!b1 && !b2)
		{
			PORTA = 0x00;
		}

		// if  PB0 button  pressed then UP counter
		else if(b1 && !b2)
		{
			for(int i=0;i<=9;i++)
			{
				PORTA = hex[i];
				_delay_ms(100);
			}
		}

		//if  PD5 button pressed then DOWN counter
		else if(!b1 && b2)
		{
			for(int i=9;i>=0;i--)
			{
				PORTA = hex[i];
				_delay_ms(100);
			}
		}

		//if  Both PB0 and PD5  pressed then HEX counter 0 to F
		else if(b1 && b2)
		{
			for(int i=0;i<16;i++)
			{
				PORTA = hex[i];
				_delay_ms(100);
			}
		}
	}
}
